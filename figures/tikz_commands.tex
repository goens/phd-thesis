\newcommand \drawpe[2] {
  \draw ({#1+0}, {#2+0}) rectangle ++(2,2);
  \draw ({#1+2},{#2+2}) -- ++(0.5,0.5);
  \draw ({#1+2.5},{#2+2.5}) rectangle ++(1,1) ++(-0.5,-0.5) node[font=\Large] {R};
}

\newcommand \drawmesh[4] {
  \foreach \x in {0,...,{#1 - 1}}{
    \foreach \y in {0,...,{#2 - 1}}{
      \drawpe{{#3+\x*4}}{{#4+\y*4}}
    }
  }
}
